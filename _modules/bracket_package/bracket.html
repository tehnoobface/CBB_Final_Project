class Bracket:
    """Class to represent a tournament bracket.

    Attributes:
        teams (pd.DataFrame): DataFrame containing information about the teams.
        method (str): The ranking method used for the tournament.
        south, east, west, midwest, region_winner (pd.DataFrame): Empty DataFrames to store regional information.

    Methods:
        getData: Load tournament data from a specified dataset.
        getBracketTeams: Prompt the user to input the playoff bracket year and initialize regional teams.
        getMethod: Prompt the user to select a ranking method for the tournament.
        generateBracket: Simulate and print tournament rounds for a given region and ranking method.
        simulateOverallWinner: Simulate and print the overall winner of the tournament.
    """
    def __init__(self):
        """Initialize a Bracket object."""
        self.teams = self.getData()
        self.method = self.getMethod()

        self.south = pd.DataFrame()
        self.east = pd.DataFrame()
        self.west = pd.DataFrame()
        self.midwest = pd.DataFrame()
        self.region_winner = pd.DataFrame()

    def generateBracket(self, region_teams, method):
        """Simulate and print tournament rounds for a given region and ranking method.

        Args:
            region_teams (pd.DataFrame): DataFrame containing regional teams.
            method (str): The ranking method used for the tournament.
        """
        num_teams = len(region_teams)
        num_rounds = int(4)

        for round_num in range(num_rounds):
            round_teams = int(num_teams / (2 ** (round_num + 1)))
            spacing = " " * (2 ** (num_rounds + round_num - 1))
            winners = []
            winner = []
            print(f"Round {round_num + 1}:")

            for i in range(round_teams):
                # team1 is the first team in the TEAM column of region_teams:
                team1 = region_teams.iloc[i * 2]
                team2 = region_teams.iloc[i * 2 + 1]
                method = str(method)

                if method == 'W' or method == 'BARTHAG':
                    winner = team1 if team1[method] > team2[method] else team2
                elif method == 'RK' or method == 'SEED':
                    winner = team1 if team1[method] < team2[method] else team2
                else:
                    print("Invalid method. Try again.")
                    self.getMethod()
                winners.append(winner)
                print(f"{spacing}{team1['TEAM']} vs {team2['TEAM']} -> Winner: {winner['TEAM']}")

            region_teams = pd.DataFrame(winners)

        final_winner = region_teams.iloc[0]['TEAM']
        print(f"{final_winner} advances to the Final Four!")

    bracket.generateBracket(bracket.midwest, bracket.method)
    bracket.simulateOverallWinner(bracket.method)
